def getMaxProfit(pnl, k):
#get the length of the array pnl
n=len(pnl)
temp = 0
#sum of first k elements
for i in range(k):
temp += pnl[i]
out = temp
#a sliding window which sums the next next
#windows sum and compares with earlier windows sum
#finally returns the max of the sum
for i in range(k, n):
out += pnl[i] - pnl[i-k]
temp = max(temp, out)
return temp
#driver main code
pnl = [-3,4,3,-2,2,5]
k = 4
#display the output
print(getMaxProfit(pnl, k))
